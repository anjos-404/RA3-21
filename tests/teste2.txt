START

*{ Definicao de variaveis antes do uso }*
(CMD_STORE 42 X)         *{ X : int }*
(CMD_STORE 3.14 PI)      *{ PI : real }*
(CMD_LOAD X)             *{ leitura de memoria }*

*{ Operacoes aritmeticas com todos os operadores }*
(EXPR X 8 +)             *{ int + int -> int }*
(EXPR X 5 -)
(EXPR PI 2.0 *)          *{ real * real -> real }*
(EXPR 10.0 4.0 |)        *{ divisao real -> real }*
(EXPR 17 5 /)            *{ divisao inteira -> int }*
(EXPR 17 5 %)            *{ resto -> int }*
(EXPR 2 8 ^)             *{ potenciacao -> int }*

*{ Mistura int/real (alargamento) e aninhamento crescente }*
(CMD_STORE 50 Y)
(EXPR (EXPR X Y +) (EXPR X Y -) *)
(EXPR (EXPR PI 1.0 +) (EXPR X 2 *) |)
(CMD_RES 1)              *{ resultado anterior }*

*{ Decisao composta e repeticao (condicoes do tipo bool) }*
(IFELSE (X 0 >) [(CMD_STORE 1 POS)] [(CMD_STORE 0 POS)])
(CMD_STORE 0 I)
(WHILE (I 2 <) [(CMD_STORE (EXPR I 1 +) I)])
(CMD_LOAD I)

END
